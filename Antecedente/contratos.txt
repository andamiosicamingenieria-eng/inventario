' =====================================================================
' SCRIPT MEJORADO - HOJA CONTRATO
' Ubicación: Hoja "CONTRATO" ? boton_Click
'
' CORRECCIONES APLICADAS:
'   ? Bug de ruta duplicada en PDF corregido
'   ? Unprotect/Protect centralizados (ya no se llama 15 veces)
'   ? Validaciones con GoTo en lugar de pirámide de If anidados
'   ? Folio incrementa solo si el proceso completa correctamente
'   ? Clave declarada correctamente como variable
'
' FUNCIONALIDADES NUEVAS:
'   ? Tabla vertical en hoja "Contratos" del mismo libro
'   ? Una fila por producto (Cantidad, SKU, Descripcion, Pesos)
'   ? can_items calculado automáticamente (conteo de productos)
'   ? PDF en carpeta Año\Mes\ con nombre CT-XXXX-AAMMDD.pdf
'   ? Campos obligatorios validados antes de guardar
'
' ? CONFIGURACIÓN — ajusta estas rutas según tu equipo
' =====================================================================

Private Sub boton_Click()

    ' -----------------------------------------------------------------
    ' ? AJUSTA ESTAS RUTAS SEGÚN TU EQUIPO
    '
    ' RutaGeneral : carpeta donde se organizan PDFs por Año\Mes
    ' RutaV1-V8   : carpeta de cada vendedor (una copia por vendedor)
    ' -----------------------------------------------------------------
    Dim RutaGeneral  As String
    Dim RutaV1       As String
    Dim RutaV2       As String
    Dim RutaV3       As String
    Dim RutaV4       As String
    Dim RutaV5       As String
    Dim RutaV6       As String
    Dim RutaV7       As String
    Dim RutaV8       As String

    RutaGeneral = "G:\.shortcut-targets-by-id\1M52sd_Ql6bwrWoegELuWJbM18alQlx8F\1.3.- REVISOR DE CONTRATOS\1.- CONTRATOS\"
    RutaV1 = "G:\.shortcut-targets-by-id\1M52sd_Ql6bwrWoegELuWJbM18alQlx8F\1.3.- REVISOR DE CONTRATOS\CONTRATOS V1 YAKIN\"
    RutaV2 = "G:\.shortcut-targets-by-id\1M52sd_Ql6bwrWoegELuWJbM18alQlx8F\1.3.- REVISOR DE CONTRATOS\CONTRATOS V2 FA\"
    RutaV3 = "G:\.shortcut-targets-by-id\1M52sd_Ql6bwrWoegELuWJbM18alQlx8F\1.3.- REVISOR DE CONTRATOS\CONTRATOS V3\"
    RutaV4 = "G:\.shortcut-targets-by-id\1M52sd_Ql6bwrWoegELuWJbM18alQlx8F\1.3.- REVISOR DE CONTRATOS\CONTRATOS V4 PAU\"
    RutaV5 = "G:\.shortcut-targets-by-id\1M52sd_Ql6bwrWoegELuWJbM18alQlx8F\1.3.- REVISOR DE CONTRATOS\CONTRATOS V5 VACANTE\"
    RutaV6 = "G:\.shortcut-targets-by-id\1M52sd_Ql6bwrWoegELuWJbM18alQlx8F\1.3.- REVISOR DE CONTRATOS\CONTRATOS V6 SAM\"
    RutaV7 = "G:\.shortcut-targets-by-id\1M52sd_Ql6bwrWoegELuWJbM18alQlx8F\1.3.- REVISOR DE CONTRATOS\CONTRATOS V7 HARLEY\"
    RutaV8 = "G:\.shortcut-targets-by-id\1M52sd_Ql6bwrWoegELuWJbM18alQlx8F\1.3.- REVISOR DE CONTRATOS\CONTRATOS V8 AARON\"
    ' -----------------------------------------------------------------

    Dim clave          As String
    Dim contrato       As Worksheet
    Dim contratos2024  As Worksheet
    Dim NumContrato    As Long
    Dim NombreArchivo  As String
    Dim RutaPDF        As String
    Dim fila           As Long

    clave = "2299"
    Set contrato = ThisWorkbook.Sheets("CONTRATO")
    Set contratos2024 = ThisWorkbook.Sheets("CONTRATOS 2024")

    ' --- Confirmación inicial ---
    If MsgBox("¿Quieres guardar los datos?", vbQuestion + vbYesNo, "Advertencia") = vbNo Then
        Exit Sub
    End If

    ' --- Desproteger hojas UNA sola vez ---
    contrato.Unprotect Password:=clave

    ' =================================================================
    ' VALIDACIONES DE CAMPOS OBLIGATORIOS
    ' Usando GoTo para evitar pirámide de If anidados
    ' =================================================================
    If IsEmpty(contrato.Range("B10")) Then
        MsgBox "Captura el número de Contrato (B10)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("D11")) Then
        MsgBox "Captura el Cliente o Empresa (D11)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("C12")) Then
        MsgBox "Captura la Dirección (C12)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("C13")) Then
        MsgBox "Captura el Teléfono (C13)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("K9")) Then
        MsgBox "Captura el Encargado (K9)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("A15")) Then
        MsgBox "Captura Dirección de Proyecto (A15)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("C16")) Then
        MsgBox "Captura Fecha de Pago (C16)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("F17")) Then
        MsgBox "Captura Fecha inicio de renta (F17)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("G18")) Then
        MsgBox "Captura quién recibe (G18)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("J18")) Then
        MsgBox "Captura Móvil de quien recibe (J18)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("L18")) Then
        MsgBox "Captura la Factura (L18)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("D10")) Then
        MsgBox "Captura la Sucursal (D10)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("K14")) Then
        MsgBox "Captura el Medio de Contacto (K14)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("C18")) Then
        MsgBox "Captura la Forma de pago (C18)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If contrato.Range("K52").Value <= 0 Then
        MsgBox "Captura las cantidades de cada producto (K52 debe ser > 0)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If contrato.Range("K160").Value <= 0 Then
        MsgBox "Captura un valor para el pagaré (K160)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If
    If IsEmpty(contrato.Range("K18")) Then
        MsgBox "Captura el Vendedor / Agente (K18)", vbExclamation, "Campo requerido"
        GoTo Reproteger
    End If

    ' =================================================================
    ' CONSTRUIR RUTAS Y NOMBRE DE ARCHIVO PDF
    '
    ' Formato nombre  : C-XXXX-VX-AAMMDD.pdf
    ' Ejemplo         : C-0012-V3-260219.pdf
    ' Copia 1 (General): RutaGeneral\YYYY\Mes\C-XXXX-VX-AAMMDD.pdf
    ' Copia 2 (Agente) : RutaVX\C-XXXX-VX-AAMMDD.pdf
    ' =================================================================
    NumContrato = contrato.Cells(10, 2).Value

    Dim FechaHoy     As Date
    Dim Anio         As String
    Dim NombreMes    As String
    Dim FechaStr     As String
    Dim FolioStr     As String
    Dim VendedorStr  As String
    Dim NombrePDF    As String    ' Solo el nombre del archivo sin ruta
    Dim RutaPDFGen   As String    ' Ruta carpeta General (con Año\Mes)
    Dim RutaPDFAgent As String    ' Ruta carpeta del agente
    Dim RutaAgente   As String    ' Raíz de la carpeta del vendedor

    FechaHoy = Now()
    Anio = Format(FechaHoy, "YYYY")
    FechaStr = Format(FechaHoy, "YYMMDD")
    FolioStr = Format(NumContrato, "0000")
    NombreMes = ObtenerNombreMes(Month(FechaHoy))
    VendedorStr = "V" & Trim(CStr(contrato.Cells(18, 12).Value))  ' L18: 1,2..8 ? se convierte a V1,V2..V8

    ' Nombre del archivo (igual para ambas copias)
    NombrePDF = "C-" & FolioStr & "-" & VendedorStr & "-" & FechaStr & ".pdf"

    ' Ruta carpeta General: CONTRATOS\GENERAL\YYYY\Mes\
    RutaPDFGen = RutaGeneral & Anio & "\" & NombreMes & "\"

    ' Seleccionar carpeta del agente según valor de L18
    Select Case UCase(Trim(VendedorStr))
        Case "V1": RutaAgente = RutaV1
        Case "V2": RutaAgente = RutaV2
        Case "V3": RutaAgente = RutaV3
        Case "V4": RutaAgente = RutaV4
        Case "V5": RutaAgente = RutaV5
        Case "V6": RutaAgente = RutaV6
        Case "V7": RutaAgente = RutaV7
        Case "V8": RutaAgente = RutaV8
        Case Else: RutaAgente = RutaGeneral  ' Si no coincide, va a General
    End Select

    RutaPDFAgent = RutaAgente

    ' Crear carpetas si no existen
    Call CrearCarpetaSiNoExiste(RutaPDFGen)
    Call CrearCarpetaSiNoExiste(RutaPDFAgent)

    ' Verificar si el PDF ya existe en carpeta General
    If Dir(RutaPDFGen & NombrePDF) <> "" Then
        Dim respuesta As Integer
        respuesta = MsgBox("El archivo '" & NombrePDF & "' ya existe." & vbCrLf & _
                           "Puede indicar un numero de contrato duplicado." & vbCrLf & _
                           "Deseas sobreescribirlo?", vbYesNo + vbExclamation, "Archivo existente")
        If respuesta = vbNo Then
            MsgBox "Operacion cancelada. Revisa el folio en B10.", vbInformation, "Cancelado"
            GoTo Reproteger
        End If
    End If

    ' =================================================================
    ' GUARDAR EN "CONTRATOS 2024" (proceso horizontal original)
    ' =================================================================
    fila = 3
    Do While contratos2024.Cells(fila, "A").Value <> ""
        fila = fila + 1
    Loop

    contratos2024.Cells(fila, 1).Value = NumContrato
    contratos2024.Cells(fila, 2).Value = contrato.Cells(18, 12).Value   ' Vendedor    (L18)
    contratos2024.Cells(fila, 3).Value = contrato.Cells(10, 6).Value    ' Ren. ant.   (F10)
    contratos2024.Cells(fila, 4).Value = contrato.Cells(10, 8).Value    ' Ren. post.  (H10)
    contratos2024.Cells(fila, 5).Value = contrato.Cells(9, 11).Value    ' Fecha       (K9)
    contratos2024.Cells(fila, 6).Value = contrato.Cells(11, 4).Value    ' Razón       (D11)
    contratos2024.Cells(fila, 7).Value = contrato.Cells(17, 3).Value    ' Días renta  (C17)
    contratos2024.Cells(fila, 8).Value = contrato.Cells(17, 6).Value    ' Inicio      (F17)
    contratos2024.Cells(fila, 10).Value = IIf(CheckBox1.Value, contrato.Cells(10, 10).Value, "EN RENTA")
    contratos2024.Cells(fila, 11).Value = contrato.Cells(63, 11).Value  ' Subtotal    (K63)
    contratos2024.Cells(fila, 15).Value = contrato.Cells(18, 7).Value   ' Día pago    (G18)
    contratos2024.Cells(fila, 19).Value = contrato.Cells(18, 10).Value  ' Factura     (J18)
    contratos2024.Cells(fila, 20).Value = contrato.Cells(18, 3).Value   ' Método pago (C18)
    contratos2024.Cells(fila, 25).Value = contrato.Cells(10, 4).Value   ' Sucursal    (D10)
    contratos2024.Cells(fila, 34).Value = contrato.Cells(14, 11).Value  ' Med.Contacto(K14)
    contratos2024.Cells(fila, 12).Value = contrato.Cells(64, 11).Value  ' IVA (K64)


    ' =================================================================
    ' GUARDAR HISTÓRICO VERTICAL EN HOJA "Contratos" (mismo libro)
    ' =================================================================
    Call GuardarHistoricoVerticalContratos(contrato, clave)

    ' =================================================================
    ' GENERAR PDF — DOS COPIAS: carpeta General + carpeta del Agente
    ' =================================================================
    contrato.ExportAsFixedFormat Type:=xlTypePDF, Filename:=RutaPDFGen & NombrePDF, Quality:=xlQualityStandard
    contrato.ExportAsFixedFormat Type:=xlTypePDF, Filename:=RutaPDFAgent & NombrePDF, Quality:=xlQualityStandard

    MsgBox "PDF guardado en dos ubicaciones:" & vbCrLf & _
           "  General : GENERAL\" & Anio & "\" & NombreMes & "\" & NombrePDF & vbCrLf & _
           "  Agente  : AGENTES\" & VendedorStr & "\" & NombrePDF, vbInformation, "Exito"

    ' =================================================================
    ' LIMPIAR CAMPOS
    ' Desproteger explícitamente aquí — las subrutinas anteriores
    ' pueden haber re-protegido la hoja como efecto secundario
    ' =================================================================
    contrato.Unprotect Password:=clave

    ' Las propiedades Locked solo funcionan con hoja desprotegida.
    ' En lugar de cambiar Locked (que puede fallar), simplemente
    ' limpiamos directamente — la hoja ya está desprotegida.
    contrato.Range("J18").ClearContents
    contrato.Range("L18").ClearContents
    contrato.Range("K160").ClearContents
    contrato.Range("D11:G11").ClearContents
    contrato.Range("A15:G15").ClearContents
    contrato.Range("C16").ClearContents
    contrato.Range("C17").ClearContents
    On Error Resume Next                     ' F17 puede tener validacion de datos
    contrato.Range("F17").Value = ""         ' Value="" en lugar de ClearContents
    contrato.Range("F17").ClearContents      ' Segundo intento si el primero no limpia
    On Error GoTo 0
    contrato.Range("G18").ClearContents
    contrato.Range("A21:B51").ClearContents
    contrato.Range("C21:D51").ClearContents
    contrato.Range("F10").ClearContents
    contrato.Range("H10").ClearContents
    contrato.Range("C18").ClearContents
    contrato.Range("F16").ClearContents
    contrato.Range("K14:L14").ClearContents
    contrato.Range("D55:D60").ClearContents
    contrato.Range("D63").ClearContents
    contrato.Range("C54:D54").Value = 0
    contrato.Range("K75").Value = 0
    CheckBox3.Value = False

    ' Incrementar folio DESPUÉS de que todo salió bien
    NumContrato = NumContrato + 1
    contrato.Cells(10, 2).Value = NumContrato

    ThisWorkbook.Save

    MsgBox "Contrato " & (NumContrato - 1) & " guardado correctamente.", vbInformation, "Completado"

Reproteger:
    contrato.Protect Password:=clave

End Sub


' =====================================================================
' FUNCIÓN AUXILIAR: Elimina caracteres inválidos para nombres de archivo
' Windows no permite: \ / : * ? " < > |
' También reemplaza espacios por guión bajo para mayor compatibilidad
' =====================================================================
Private Function LimpiarNombreArchivo(ByVal texto As String) As String
    ' Elimina caracteres invalidos para nombres de archivo en Windows
    texto = Replace(texto, "\", "")
    texto = Replace(texto, "/", "")
    texto = Replace(texto, ":", "")
    texto = Replace(texto, "*", "")
    texto = Replace(texto, "?", "")
    texto = Replace(texto, Chr(34), "")  ' Comilla doble usando Chr() para evitar conflicto
    texto = Replace(texto, "<", "")
    texto = Replace(texto, ">", "")
    texto = Replace(texto, "|", "")
    texto = Replace(texto, " ", "_")     ' Espacios a guion bajo
    LimpiarNombreArchivo = texto
End Function


' =====================================================================
' FUNCIÓN AUXILIAR: Nombre del mes en español
' =====================================================================
Private Function ObtenerNombreMes(ByVal numMes As Integer) As String
    Dim meses(1 To 12) As String
    meses(1) = "Enero": meses(2) = "Febrero": meses(3) = "Marzo"
    meses(4) = "Abril": meses(5) = "Mayo": meses(6) = "Junio"
    meses(7) = "Julio": meses(8) = "Agosto": meses(9) = "Septiembre"
    meses(10) = "Octubre": meses(11) = "Noviembre": meses(12) = "Diciembre"
    ObtenerNombreMes = meses(numMes)
End Function


' =====================================================================
' SUB AUXILIAR: Crea toda la cadena de carpetas nivel por nivel
' =====================================================================
Private Sub CrearCarpetaSiNoExiste(ByVal ruta As String)
    Dim partes()  As String
    Dim acumulada As String
    Dim i         As Integer

    If Right(ruta, 1) = "\" Then ruta = Left(ruta, Len(ruta) - 1)
    partes = Split(ruta, "\")
    acumulada = partes(0)

    For i = 1 To UBound(partes)
        acumulada = acumulada & "\" & partes(i)
        If Dir(acumulada, vbDirectory) = "" Then
            On Error Resume Next
            MkDir acumulada
            On Error GoTo 0
        End If
    Next i
End Sub


' =====================================================================
' SUB AUXILIAR: Histórico Vertical en hoja "Contratos" del mismo libro
'
' Una fila por producto. Columnas:
'   A=Contrato   B=Cliente     C=Obra        D=Ren_Anterior  E=Ren_Posterior
'   F=Importe_c  G=Dias_renta  H=Fecha       I=Telefono      J=Encargado
'   K=can_items  L=SKU         M=Descripcion N=Peso_unitario O=Peso_total
'   P=Cantidad   Q=Fecha Registro
' =====================================================================
Private Sub GuardarHistoricoVerticalContratos(ByRef contrato As Worksheet, ByVal clave As String)

    Dim wsContratos  As Worksheet
    Dim lastRowVert  As Long
    Dim i            As Integer
    Dim canItems     As Integer
    Dim wsBuscar     As Worksheet
    Dim hojaExiste   As Boolean

    ' Buscar hoja "Contratos" — distinta de "CONTRATO" (hoja de captura)
    hojaExiste = False
    For Each wsBuscar In ThisWorkbook.Sheets
        If UCase(Trim(wsBuscar.Name)) = "CONTRATOS" Then
            Set wsContratos = wsBuscar
            hojaExiste = True
            Exit For
        End If
    Next wsBuscar

    ' Si no existe, crearla y agregar encabezados
    If Not hojaExiste Then
        Set wsContratos = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        wsContratos.Name = "Contratos"

        With wsContratos
            .Cells(1, 1).Value = "Contrato"
            .Cells(1, 2).Value = "Cliente"
            .Cells(1, 3).Value = "Obra"
            .Cells(1, 4).Value = "Ren_Anterior"
            .Cells(1, 5).Value = "Ren_Posterior"
            .Cells(1, 6).Value = "Importe_c"
            .Cells(1, 7).Value = "Dias_renta"
            .Cells(1, 8).Value = "Fecha"
            .Cells(1, 9).Value = "Telefono"
            .Cells(1, 10).Value = "Encargado"
            .Cells(1, 11).Value = "can_items"
            .Cells(1, 12).Value = "SKU"
            .Cells(1, 13).Value = "Descripcion"
            .Cells(1, 14).Value = "Peso_unitario"
            .Cells(1, 15).Value = "Peso_total"
            .Cells(1, 16).Value = "Cantidad"
            .Cells(1, 17).Value = "Tipo"
            .Cells(1, 18).Value = "Fecha_Registro"

            With .Range("A1:R1")
                .Font.Bold = True
                .Interior.Color = RGB(31, 78, 121)
                .Font.Color = RGB(255, 255, 255)
                .HorizontalAlignment = xlCenter
            End With
            .Columns("A:R").AutoFit
        End With
    End If

    wsContratos.Unprotect Password:=clave

    ' Leer datos de encabezado
    Dim NumContrato   As Long
    Dim Cliente       As String
    Dim Obra          As String
    Dim RenAnterior   As Variant
    Dim RenPosterior  As Variant
    Dim ImporteC      As Variant
    Dim DiasRenta     As Variant
    Dim Fecha         As Variant
    Dim Telefono      As String
    Dim Encargado     As String

    NumContrato = CLng(contrato.Cells(10, 2).Value)        ' B10
    Cliente = CStr(contrato.Cells(11, 3).Value)            ' D11
    Obra = CStr(contrato.Cells(10, 3).Value)               ' C10
    RenAnterior = contrato.Cells(10, 6).Value              ' F10
    RenPosterior = contrato.Cells(10, 8).Value             ' H10
    ImporteC = contrato.Cells(62, 4).Value                 ' D62
    DiasRenta = contrato.Cells(52, 1).Value                ' A52
    Fecha = contrato.Cells(16, 3).Value                    ' C16
    Telefono = CStr(contrato.Cells(16, 7).Value)           ' G16
    Encargado = CStr(contrato.Cells(9, 11).Value)          ' K9

    Dim Tipo As String
    Tipo = IIf(CheckBox1.Value, CStr(contrato.Cells(10, 10).Value), "EN RENTA")  ' misma logica que historico horizontal

    ' Contar productos (can_items)
    canItems = 0
    For i = 21 To 51
        If contrato.Cells(i, 1).Value <> "" Then canItems = canItems + 1
    Next i

    ' Escribir una fila por producto
    For i = 21 To 51
        If contrato.Cells(i, 1).Value <> "" Then

            lastRowVert = wsContratos.Cells(wsContratos.Rows.Count, 1).End(xlUp).Row + 1

            wsContratos.Cells(lastRowVert, 1).Value = NumContrato
            wsContratos.Cells(lastRowVert, 2).Value = Cliente
            wsContratos.Cells(lastRowVert, 3).Value = Obra
            wsContratos.Cells(lastRowVert, 4).Value = RenAnterior
            wsContratos.Cells(lastRowVert, 5).Value = RenPosterior
            wsContratos.Cells(lastRowVert, 6).Value = ImporteC
            wsContratos.Cells(lastRowVert, 7).Value = DiasRenta
            wsContratos.Cells(lastRowVert, 8).Value = Fecha
            wsContratos.Cells(lastRowVert, 9).Value = Telefono
            wsContratos.Cells(lastRowVert, 10).Value = Encargado
            wsContratos.Cells(lastRowVert, 11).Value = canItems
            wsContratos.Cells(lastRowVert, 12).Value = CStr(contrato.Cells(i, 3).Value)
            wsContratos.Cells(lastRowVert, 13).Value = CStr(contrato.Cells(i, 5).Value)
            wsContratos.Cells(lastRowVert, 14).Value = contrato.Cells(i, 9).Value
            wsContratos.Cells(lastRowVert, 15).Value = contrato.Cells(i, 11).Value
            wsContratos.Cells(lastRowVert, 16).Value = contrato.Cells(i, 1).Value
            wsContratos.Cells(lastRowVert, 17).Value = Tipo
            wsContratos.Cells(lastRowVert, 18).Value = Now()

            If (lastRowVert Mod 2) = 0 Then
                wsContratos.Range("A" & lastRowVert & ":R" & lastRowVert).Interior.Color = RGB(189, 215, 238)
            Else
                wsContratos.Range("A" & lastRowVert & ":R" & lastRowVert).Interior.Color = RGB(255, 255, 255)
            End If

        End If
    Next i
    
    wsContratos.Columns("A:R").AutoFit
    wsContratos.Protect Password:=clave

End Sub
